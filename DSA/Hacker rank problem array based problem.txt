package main

import "fmt"


func birthdayCakeCandles(candles []int32) int32 {
  
   var max int32
    var count int32
    for _, v := range candles {

        if v > max {

            max = v
            count = 1
        } else if v == max {

            count++
        }

    }

    return count
}


func main (){

  
    arr:=[]int32{4,1,3,4}
    get := birthdayCakeCandles(arr)
    fmt.Println(get)

}
