package main

import (
	"fmt"
)

func main() {

	var n = []int{1, 39, 2, 9, 7, 54, 11}

	var isdone = false

	for !isdone {

		isdone = true

		i := 0

		for i < len(n)-1 {

			if n[i] > n[i+1] {

				n[i], n[i+1] = n[i+1], n[i]
				isdone = false
			}
			i++
		}

	}
	fmt.Println(n)
}
